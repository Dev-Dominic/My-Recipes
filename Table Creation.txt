Create Table User(
	id varchar(255) not null,
	first_name varchar(30),
	last_name varchar(30),
	email varchar(40),
	password varchar(40),
	primary key(id)
);
Create Table Ingredients(
	id varchar(255) not null,
	name varchar(30),
	avaiable_quantity varchar(50),
	user_id varchar(15),
	measurement_name(30),
	primary key(id),
	foreign key(user_id)
);

Create Table Recipe(
	id varchar(200),
	name varchar(100),
	instructions varchar(30),
	date_created date,
	user_id varchar(15),
	ingredient_id varchar(255),
	primary key(recipe_id),
	foreign key(user_id)
);


Create Table Recipe_Ingredients(
	recipe_id varchar(200),
	ingredient_id varchar(255),
	foreign key(recipe_id),
	foreign key(ingredient_id)
);


Create Table Meal(
	id varchar(15),
	image varbinary(max),
	calories varchar(30),
	servings varchar(30),
	primary key(id)
);

Create Table Measurements(
	id varchar(15),
	unit varbinary(max),
	amount int,
	primary key(id)
);

Create Table Planned_Meal(
	id varchar(15),
	week_start date,
	week_start date,
	primary key(id)
);
